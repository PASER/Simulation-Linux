00047 void PASER_Route_Discovery::route_discovery(struct in_addr dest_addr,
00048         int isDestGW) {
00049     // If a route discovery has been already started for dest_addr,
00050     // then simply return
00051     struct in_addr bcast_addr;
00052     bcast_addr.S_addr = (in_addr_t) 0xFFFFFFFF;
00053     if (paser_global->getRreq_list()->pending_find(dest_addr)) {
00054         EV << "a route discovery are already started\n";
00055         return;
00056     }
00057 
00058     PASER_UB_RREQ *message = NULL;
00059     if (isDestGW) {
00060         paser_global->generateGwSearchNonce();
00061     }
00062     for (u_int32_t i = 0; i < paser_configuration->getNetDeviceNumber(); i++) {
00063         in_addr WlanAddrStruct;
00064         WlanAddrStruct.S_addr = paser_global->getNetDevice()[i].ipaddr.S_addr;
00065         if (message != NULL) {
00066             delete message;
00067         }
00068         message = paser_global->getPaket_processing()->send_ub_rreq(
00069                 WlanAddrStruct, dest_addr, isDestGW);
00070     }
00071     paser_global->setSeqNr(paser_global->getSeqNr() + 1);
00072 
00073     if (message == NULL)
00074         return;
00075     // Record information for destination
00076     message_rreq_entry *pend_rreq = paser_global->getRreq_list()->pending_add(
00077             dest_addr);
00078 
00079     PASER_Timer_Message *tPack = new PASER_Timer_Message();
00080     tPack->data = (void *) message;
00081     tPack->destAddr.S_addr = dest_addr.S_addr;
00082     tPack->handler = ROUTE_DISCOVERY_UB;
00083     struct timeval now;
00084     paser_modul->MYgettimeofday(&now, NULL);
00085     tPack->timeout = timeval_add(now, PASER_UB_RREQ_WAIT_TIME);
00086     pend_rreq->tries = 0;
00087 
00088     ev << "now: " << now.tv_sec << "\ntimeout: " << tPack->timeout.tv_sec
00089             << "\n";
00090     paser_global->getTimer_queue()->timer_add(tPack);
00091     pend_rreq->tPack = tPack;
00092 }
