00055 int PASER_Timer_Queue::timer_remove(PASER_Timer_Message *t) {
00056     if (!t) {
00057         ev << "timer == NULL\n";
00058         return 0;
00059     }
00060     if (t->handler == KDC_REQUEST) {
00061         for (std::list<PASER_Timer_Message *>::iterator it = timer_queue.begin();
00062                 it != timer_queue.end(); it++) {
00063             PASER_Timer_Message *temp = (PASER_Timer_Message *) *it;
00064             if (temp->handler == KDC_REQUEST) {
00065                 timer_queue.erase(it);
00066                 delete temp;
00067                 return 1;
00068             }
00069         }
00070     } else {
00071         for (std::list<PASER_Timer_Message *>::iterator it = timer_queue.begin();
00072                 it != timer_queue.end(); it++) {
00073             PASER_Timer_Message *temp = (PASER_Timer_Message *) *it;
00074             if (temp->handler == ROUTINGTABLE_DELETE_ENTRY
00075                     || temp->handler == ROUTINGTABLE_VALID_ENTRY
00076                     || temp->handler == NEIGHBORTABLE_DELETE_ENTRY
00077                     || temp->handler == NEIGHBORTABLE_VALID_ENTRY) {
00078                 if (temp->destAddr.S_addr == t->destAddr.S_addr
00079                         && temp->handler == t->handler) {
00080                     timer_queue.erase(it);
00081                     return 1;
00082                 }
00083             } else {
00084                 if (temp->destAddr.S_addr == t->destAddr.S_addr
00085                         && temp->handler == t->handler) {
00086                     timer_queue.erase(it);
00087                     return 1;
00088                 }
00089             }
00090         }
00091         ev << "timer not found\n";
00092         return 0;
00093     }
00094 }
