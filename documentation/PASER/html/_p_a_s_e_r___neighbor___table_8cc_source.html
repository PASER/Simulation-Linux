00121 void PASER_Neighbor_Table::updateNeighborTableAndSetTableTimeout(
00122         struct in_addr neigh, int nFlag, u_int8_t *root, int iv,
00123         geo_pos position, X509 *cert, struct timeval now, u_int32_t ifIndex) {
00124     PASER_Timer_Message *deletePack = NULL;

00126     PASER_Neighbor_Entry *nEntry = findNeigh(neigh);
00127     int32_t setTrusted = 0;
00128     if (nEntry) {
00129         setTrusted = nEntry->neighFlag;
00130         deletePack = nEntry->deleteTimer;
00131         validPack = nEntry->validTimer;
00132         if (validPack == NULL) {
00133             validPack = new PASER_Timer_Message();
00134             validPack->data = NULL;
00135             validPack->destAddr.S_addr = neigh.S_addr;
00136             validPack->handler = NEIGHBORTABLE_VALID_ENTRY;
00137             nEntry->validTimer = validPack;
00138             validPack->timeout = timeval_add(now, PASER_NEIGHBOR_VALID_TIME);
00139             setTrusted = 0;
00140         } else if (nFlag) {
00141             validPack->timeout = timeval_add(now, PASER_NEIGHBOR_VALID_TIME);
00142         }
00143         deletePack->timeout = timeval_add(now, PASER_NEIGHBOR_DELETE_TIME);
00144     } else {
00145         setTrusted = nFlag;
00146         deletePack = new PASER_Timer_Message();
00147         deletePack->data = NULL;
00148         deletePack->destAddr.S_addr = neigh.S_addr;
00149         deletePack->handler = NEIGHBORTABLE_DELETE_ENTRY;
00150         validPack = new PASER_Timer_Message();
00151         validPack->data = NULL;
00152         validPack->destAddr.S_addr = neigh.S_addr;
00153         validPack->handler = NEIGHBORTABLE_VALID_ENTRY;
00154         deletePack->timeout = timeval_add(now, PASER_NEIGHBOR_DELETE_TIME);
00155         validPack->timeout = timeval_add(now, PASER_NEIGHBOR_VALID_TIME);
00156     }
00157 
00158     if (validPack != 0) {
00159         ev << "Route valid timeout: " << validPack->timeout.tv_sec << "\n";
00160         timer_queue->timer_add(validPack);
00161     }
00162     ev << "now: " << now.tv_sec << "\nRoute delete timeout: "
00163             << deletePack->timeout.tv_sec << "\n";
00164     timer_queue->timer_add(deletePack);
00165 
00166     u_int8_t *rootN = (u_int8_t *) malloc(
00167             (sizeof(u_int8_t) * SHA256_DIGEST_LENGTH));
00168     memcpy(rootN, root, (sizeof(u_int8_t) * SHA256_DIGEST_LENGTH));
00169     if (nEntry == NULL) {
00170         nEntry = insert(neigh, deletePack, validPack, nFlag, rootN, iv,
00171                 position, (u_int8_t*) cert, ifIndex);
00172     } else {
00173         int tempFlag = 0;
00174         if (nEntry->neighFlag || nFlag) {
00175             tempFlag = 1;
00176         }
00177         nEntry = update(nEntry, neigh, deletePack, validPack, tempFlag, rootN,
00178                 iv, position, (u_int8_t*) cert, ifIndex);
00179     }
00180     nEntry->neighFlag = setTrusted || nFlag;
